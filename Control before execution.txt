setprefix.MouseButton1Click:Connect(function()
    if prefixbox.Text ~= "" then
        guiprefix = prefixbox.Text
        prefixlabel.Text = "Set Prefix to "..guiprefix
        wait(2)
        prefixlabel.Text = "Prefix"
        else
        prefixlabel.Text = "Error: Prefix cant be empty"
        wait(3)
        prefixlabel.Text = "Prefix"
        end
end)

setname.MouseButton1Click:Connect(function()
    if namebox.Text ~= "" then

        guinameshow = namebox.Text

        wait(2)
        namelabel.Text = "Name"

        wait(3)
        namelabel.Text = "Name"
        end
end)
